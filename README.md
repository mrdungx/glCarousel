glCarousel
==========

Usage:

  		$('#carousel-1').glCarousel();

			$('#carousel-2').glCarousel({
				itemsToScroll: 1,
				scrollTime: 800,
				loop: true,
				interval: 0,
				pagination: true
			});

			$('#carousel-3').glCarousel({
				itemsToScroll: 2,
				scrollTime: 800,
				loop: true,
				interval: 3000,
				pagination: false
			});
